resp=rnorm(100,100,5)
trat=rep(c("A","B"),e=50)
test_two(trat,resp)

## t = 0.76294, df = 94.567, p-value = 0.4474

## 95 percent confidence interval:

## sample estimates:
## mean in group A mean in group B
